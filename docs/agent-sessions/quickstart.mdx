## Prerequisites

Set two server-side environment variables:

```bash
OPENCOMPUTER_API_KEY=osb_...
ANTHROPIC_API_KEY=sk-ant-...
```

<Tip>Grab your OpenComputer API key from [app.opencomputer.dev](https://app.opencomputer.dev). Sessions run the model on your Anthropic key.</Tip>

## 1. Add the server helper

Keep all management calls server-side. This helper creates the reusable agent, starts sessions, and mints fresh browser tokens.

```ts
// src/oc.ts
const OC = process.env.OC_API_URL ?? "https://api.opencomputer.dev";
const ORG_KEY = process.env.OPENCOMPUTER_API_KEY;
const ANTHROPIC_KEY = process.env.ANTHROPIC_API_KEY;

const PROMPT = `
You are a coding agent working in an OpenComputer sandbox.

Use bash/read/write/ls for all filesystem and command work.
Emit short progress updates with say as you work.
Ask only when blocked. Otherwise make a reasonable choice and continue.
When done, say exactly what changed and how to verify it.
`.trim();

export async function oc<T>(path: string, init: RequestInit = {}): Promise<T> {
  if (!ORG_KEY) throw new Error("OPENCOMPUTER_API_KEY is not set");

  const res = await fetch(`${OC}/v3${path}`, {
    ...init,
    headers: {
      Authorization: `Bearer ${ORG_KEY}`,
      "Content-Type": "application/json",
      ...init.headers,
    },
  });

  if (!res.ok) {
    throw new Error(`OpenComputer ${res.status}: ${await res.text()}`);
  }

  return res.json() as Promise<T>;
}

let agentId: Promise<string> | undefined;

export function ensureAgent() {
  if (!ANTHROPIC_KEY) throw new Error("ANTHROPIC_API_KEY is not set");

  agentId ??= oc<{ id: string }>("/agents", {
    method: "POST",
    body: JSON.stringify({
      name: "quickstart-coder",
      runtime: "claude",
      model: "anthropic/claude-opus-4-8",
      prompt: PROMPT,
      key: ANTHROPIC_KEY,
      limits: { turns: 4, turn_seconds: 600 },
    }),
  })
    .then((agent) => agent.id)
    .catch((err) => {
      agentId = undefined;
      throw err;
    });

  return agentId;
}

export async function startSession(input: string) {
  const agent = await ensureAgent();

  return oc<{
    session: { id: string; status: string; head: number };
    client_token: string;
  }>("/sessions", {
    method: "POST",
    body: JSON.stringify({ agent, input }),
  });
}

export async function mintClientToken(sessionId: string) {
  return oc<{ client_token: string; expires_at: string }>(
    `/sessions/${sessionId}/client-tokens`,
    {
      method: "POST",
      body: JSON.stringify({ scopes: ["read", "steer"], ttl: 3600 }),
    },
  );
}
```

## 2. Start a session from your backend

In a Next.js route, accept a user task and return only the session plus its scoped token. The org key never reaches the browser.

```ts
// app/api/sessions/route.ts
import { startSession } from "@/oc";

export async function POST(req: Request) {
  const { input } = await req.json();

  if (typeof input !== "string" || !input.trim()) {
    return Response.json({ error: "input is required" }, { status: 400 });
  }

  try {
    return Response.json(await startSession(input));
  } catch (err) {
    return Response.json({ error: (err as Error).message }, { status: 502 });
  }
}
```

## 3. Stream events in the browser

Use the `client_token` with native `EventSource`. Ask for `progress` when you want a good user-facing activity feed, or `internal` for a debugging console.

```tsx
const OC = process.env.NEXT_PUBLIC_OC_API_URL ?? "https://api.opencomputer.dev";

type SessionEvent = {
  id: string;
  seq: number;
  type: string;
  level: "user" | "progress" | "internal";
  body?: Record<string, unknown>;
};

async function createSession(input: string) {
  const res = await fetch("/api/sessions", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ input }),
  });

  if (!res.ok) throw new Error(await res.text());
  return res.json() as Promise<{
    session: { id: string };
    client_token: string;
  }>;
}

function streamSession(
  sessionId: string,
  clientToken: string,
  onEvent: (event: SessionEvent) => void,
) {
  const url = new URL(`${OC}/v3/sessions/${sessionId}/events`);
  url.searchParams.set("stream", "sse");
  url.searchParams.set("level", "progress");
  url.searchParams.set("token", clientToken);

  const es = new EventSource(url.toString());
  es.onmessage = (message) => onEvent(JSON.parse(message.data));
  es.onerror = () => {
    // EventSource reconnects automatically. Each event seq is sent as the SSE id,
    // so the browser resumes with Last-Event-ID after a dropped connection.
  };

  return () => es.close();
}
```

## 4. Render by event type

The stable discriminator is `type`. Do not parse prose to decide whether a turn finished or failed.

```tsx
function EventRow({ event }: { event: SessionEvent }) {
  const body = event.body ?? {};

  switch (event.type) {
    case "agent.message":
      return <p>{String(body.text ?? "")}</p>;

    case "tool.call":
      return <code>$ {String(body.args_summary ?? body.tool ?? "tool")}</code>;

    case "exec.completed":
      return (
        <pre>
          exit {String(body.exit_code)}
          {"\n"}
          {String(body.summary ?? "")}
        </pre>
      );

    case "turn.completed": {
      const reason = String(body.yield_reason ?? "completed");
      if (reason === "completed") return <p>Done</p>;
      if (reason === "needs_input") return <p>Waiting for input</p>;
      return <p>Stopped: {reason}</p>;
    }

    default:
      return body.text || body.summary ? <p>{String(body.text ?? body.summary)}</p> : null;
  }
}
```

## 5. Steer the session

Post follow-up messages directly from the browser with the same `client_token`. The session wakes if it was idle and continues with its prior context.

```ts
async function steer(sessionId: string, clientToken: string, text: string) {
  const res = await fetch(`${OC}/v3/sessions/${sessionId}/messages`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${clientToken}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      text,
      idempotency_key: crypto.randomUUID(),
    }),
  });

  if (!res.ok) throw new Error(await res.text());
  return res.json();
}
```

## Production checklist

- Add app auth before minting or returning a `client_token`.
- Set `limits` on the agent or session to bound tokens, wall-clock time, and auto-run count.
- Render every non-`completed` `yield_reason` explicitly, especially `error`, `deadline_exceeded`, `budget_exceeded`, and `canceled`.
- Use `idempotency_key` for every user action you might retry.
- Use `level=user` for customer-facing chat, `level=progress` for activity feeds, and `level=internal` only for operator/debug views.
- Register a [webhook](/agent-sessions/webhooks) when your backend needs push delivery after the tab closes.
